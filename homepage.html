# Create three separate zip files for index.html, consulting.html, and book.html
# Each will include its corresponding HTML file and the CNAME file for www.newfinancialworld.com

import zipfile
from pathlib import Path

root = Path("/mnt/data")
cname_path = root / "CNAME"
if not cname_path.exists():
    cname_path.write_text("www.newfinancialworld.com\n", encoding="utf-8")

# Function to zip individual files
def make_zip(html_filename):
    html_path = root / html_filename
    zip_name = html_filename.replace(".html", ".zip")
    zip_path = root / zip_name
    with zipfile.ZipFile(zip_path, "w", zipfile.ZIP_DEFLATED) as z:
        z.write(html_path, arcname=html_filename)
        z.write(cname_path, arcname="CNAME")
    return str(zip_path)

index_zip = make_zip("index.html")
consulting_zip = make_zip("consulting.html")
book_zip = make_zip("book.html")

index_zip, consulting_zip, book_zip
